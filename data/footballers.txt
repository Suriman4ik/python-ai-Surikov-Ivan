# Запрос: Футболисты, их клубы, нации и титулы
# 
# Что делает этот запрос:
# - Находит всех футболистов (люди с профессией "футболист")
# - Для каждого футболиста показывает:
#   * Клуб(ы), за который он играет/играл
#   * Страну/нацию (гражданство)
#   * Титулы/награды, которые он выиграл
#
# Проблема: один игрок может появляться много раз (несколько клубов × несколько титулов)
# Решение: SELECT DISTINCT убирает полные дубликаты, но не группирует игроков
#
# Для подсчета уникальных игроков в pandas используйте:
# df['player'].nunique() - считает количество уникальных значений в колонке 'player'
#
SELECT DISTINCT ?player ?playerLabel ?club ?clubLabel ?countryLabel ?title ?titleLabel
WHERE {
  # Игрок: футболист
  ?player wdt:P106 wd:Q937857;    # футболист
          wdt:P27 ?country;       # страна/нация
          wdt:P54 ?club;          # футбольный клуб (членство в команде)
          wdt:P166 ?title.        # полученная награда/титул

  # Получаем метки на русском, английском, любом другом языке
  SERVICE wikibase:label { bd:serviceParam wikibase:language "ru,en,mul". }
}
# LIMIT убран - получаем все данные
